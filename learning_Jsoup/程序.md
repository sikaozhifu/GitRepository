package com.example;

import java.io.IOException;

import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;

public class JsoupTest {

	public static void main(String[] args) {
		try {
			String userAgent="Mozilla/5.0 (Windows NT 6.1; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/67.0.3396.99 Safari/537.36";
			Document doc = Jsoup.connect("https://www.zhihu.com/explore/recommendations")
				.userAgent(userAgent)
				.get();
			Element main = doc.getElementById("zh-recommend-list-full");
			Elements url = main.select("div").select("div:nth-child(1)").select("h2").select("a[class=question_link]");
			for (Element question : url) {
				String URL = question.attr("abs:href");
				Document doc2 = Jsoup.connect(URL)
					.userAgent(userAgent)
					.get();
				Elements title = doc2.getElementsByClass("QuestionHeader-title");
				Elements detail = doc2.getElementsByClass("RichText ztext");
				Elements answer = doc2.getElementsByClass("RichText ztext CopyrightRichText-richText");
				System.out.println(URL);
				System.out.println("title:"+"\n"+title.text());
				System.out.println("detail:"+"\n"+detail.text());
				System.out.println("answer:"+"\n"+answer.text());
				System.out.println("--------------------------");
			}
		} catch (IOException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}
}

title:
电气工程及其自动化专业考研高校及二级学科方向应该如何选择？
detail:
普通院校的电气工程及其自动化专业学生，面临考研选择，寻求高校推荐以及二级学科的分析，先谢过！
answer:
首先，题主要搞清楚国内电气工程学科精确的二级学科分类应为： 1.电力系统及其自动化 俗称发电、电自。通俗来说包含一次侧和二次侧两个方面，一次侧包含电力系统规划和可靠性分析，输配电方式和电力系统运行稳定性分析，新能源接入电网，电力市场等电力系统本身的问题；二次侧包含继电保护及其自动化装置，设备运行与分析控制，电力系统通讯，电力系统自动化等等电力系统设备方面的问题 2.高电压与绝缘技术 可以拆解为高电压技术和电气绝缘技术两个方面。 高电压技术主要研究高电压作用下各种绝缘介质性能和放电现象，高电压设备的绝缘结构设计，高电压试验和测量的设备及方法，电力系统的过电压、高电压或大电流相关分析应用和防护以及脉冲等离子体等。 电气绝缘技术的前身是电气电缆技术，现在主要研究电介质理论，电力设备状态监测与故障诊断，电气功能材料与特种绝缘技术，电力设备绝缘结构与配合，新型储能与能量转换材料等。 3.电力电子与电力传动 电力电子学以前叫工业电子学，半导体变流技术，是一门对电能进行变换和控制的技术。通俗来说就是“变流”，交直流互变变频等。它研究半导体开关器件及其应用以及电力系统中的电力电子设备，还有利用电力电子器件进行电机的智能、节能控制运行，是一门综合电气工程、控制理论和电子技术的交叉学科。从学科内部架构，宏观来看主要包含:电力电子器件技术，电力电子变换电路，电力电子系统控制技术，电力电子技术的应用，以及电力电子系统集成，结构设计与封装技术。 4.电机与电器 很多电气工程背景出身的同学都把这个学科当作一个研究方向，其实它分为电机和电器两个迥异的方向。国内也只有XJTU有足够的资源和师资将电器学科作为独立教研室分立出去（XJTU的绝缘和高压也是这样）。 电机不用多说，宏观来说研究各种常用电机/特种电机的设计和分析控制。可以分为三个层面：1.基础层：电磁场与多物理场，电工材料，电力电子技术与电机控制，电机测试技术；2.核心层：电机理论与技术；3.应用层：特殊电力系统，全电驱动系统，多电机系统，新能源系统等。 电器，主要面向电力系统中的开关电气设备及其关键技术与智能化。不同于电力电子中的“开关”器件，它研究的开关都是高电压大电流的应用背景，近年来热门的高压直流断路器就属于电器的研究与应用领域。它研究电器开关设备本体，脉冲功率与放电等离子体，电气开关设备智能化等。 5.电工理论与新技术 这个方向其实很广泛，与其他方向甚至学科都有交叉，俗称“三电”但又不局限于三电。在高校里面一般是担任电工学（电路电磁场），电子学（数模电）、电工原理教学科研任务以及一些电气测量设备与仪器研发。研究方向举例7：电磁场与微波技术，复杂电磁场仿真与计算，电力电子与新能源系统的建模与设计，现代电路的系统分析与测试诊断，电力系统规划与运行，非线性电路中的混沌现象与应用，电子测量与微机控制，模拟集成电路与高速数字电路等。
